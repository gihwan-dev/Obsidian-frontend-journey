import {createContext, Dispatch, SetStateAction, useContext, useState} from 'react'  
import reactLogo from './assets/react.svg'  
import viteLogo from '/vite.svg'  
import './App.css'  
  
const SomeContext = createContext<{  
    count: number;  
    setCount: Dispatch<SetStateAction<number>>;  
} | null>(null);  
  
const SomeAnotherCounter = () => {  
    const context = useContext(SomeContext);  
  
    if (!context) {  
        throw new Error("");  
    }  
  
    // 즉시 실행 함수로 현재 값을 캡처  
    console.log("Immediate value:", (function() {  
        return SomeContext._currentValue;  
    })());  
  
  
    const { setCount, count } = context;  
  
    return <div>  
        <h1>Counter</h1>  
        <p>Count: {count}</p>  
        <button onClick={() => setCount(count + 1)}>Increment</button>  
        <button onClick={() => setCount(count - 1)}>Decrement</button>  
  
    </div>  
}  
  
const Counter = () => {  
    const context = useContext(SomeContext);  
    const [anotherCount, setAnotherCount] = useState(20);  
  
    if (!context) {  
        throw new Error("");  
    }  
  
    // 즉시 실행 함수로 현재 값을 캡처  
    console.log("Immediate value:", (function() {  
        return SomeContext._currentValue;  
    })());  
  
  
    const { setCount, count } = context;  
  
  
    return <div>  
        <h1>Counter</h1>  
        <p>Count: {count}</p>  
        <button onClick={() => setCount(count + 1)}>Increment</button>  
        <button onClick={() => setCount(count - 1)}>Decrement</button>  
        <button onClick={() => setCount(0)}>Reset</button>  
        <button onClick={() => setCount(0)}>Reset</button>  
        <SomeContext.Provider value={{  
            count: anotherCount,  
            setCount: setAnotherCount  
        }}>  
            <SomeAnotherCounter />  
        </SomeContext.Provider>  
    </div>  
}  
  
function App() {  
  const [count, setCount] = useState(0)  
    return (  
    <>  
      <div>  
        <a href="https://vite.dev" target="_blank">  
          <img src={viteLogo} className="logo" alt="Vite logo" />  
        </a>  
        <a href="https://react.dev" target="_blank">  
          <img src={reactLogo} className="logo react" alt="React logo" />  
        </a>  
      </div>  
      <h1>Vite + React</h1>  
      <div className="card">  
        <button onClick={() => setCount((count) => count + 1)}>  
          count is {count}  
        </button>  
        <p>          Edit <code>src/App.tsx</code> and save to test HMR  
        </p>  
      </div>  
      <p className="read-the-docs">  
        Click on the Vite and React logos to learn more  
      </p>  
        <SomeContext.Provider value={{  
            count,  
            setCount  
        }}>  
            <Counter />  
        </SomeContext.Provider>  
    </>  
  )  
}  
  
export default App
